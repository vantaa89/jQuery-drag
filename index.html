class Block{
    constructor(id){
        this.blockno = id;
        this.id = "block" + id;
        $('.block-container').append('<div class="block" id="' + this.id + '"> </div>');
        $('#' + this.id).css({
            'top': randInt(100, 900),
            'left': randInt(100, 1100)
        });
        var prevOffset, curOffset;
        $('#' + this.id).draggable({
            addClasses: "false",
            drag: function(e, ui){
                prevOffset = curOffset;
                curOffset = $.extend({}, ui.offset);
                return true;
            }
        });
 
        $('.block:not(#' + this.id + ')').droppable({
            greedy: true,
            over: function(e, ui){
                ui.helper.offset(curOffset=prevOffset).trigger('mouseup');
            },
            tolerance: "touch",
        });        // i (>j)번째 블록은 j번째 블록 위로 가져다놓을 수가 있음. 수정 필요
    }
 
}
